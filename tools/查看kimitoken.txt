$sessionRoot = "$env:USERPROFILE\.kimi\sessions"

Get-ChildItem $sessionRoot -Recurse -Filter context.jsonl |
  Sort-Object LastWriteTime -Descending |
  Select-Object -First 1 |
  ForEach-Object {
    Write-Host "Session file:" $_.FullName
    Select-String -Path $_.FullName -Pattern '"role"\s*:\s*"_usage"' |
      Select-Object -Last 1 |
      ForEach-Object {
        $json = $_.Line | ConvertFrom-Json
        Write-Host "Last token_count:" $json.token_count
      }
  }
